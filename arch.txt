### LOCALE ###

# Keyboard layout
loadkeys uk

# Time and date
timedatectl set-ntp true



### DISK SETUP ###

# Verify if UEFI enabled
ls /sys/firmware/efi/efivars

# List all drives
fdisk -l

# Edit partitions
cfdisk /dev/sda1

# Format partition
mkfs.btrfs /dev/sda1
mkfs.ext4 /dev/sda2

# Mount partition
mount /dev/sda1 /mnt
mount /dev/sda2 /mnt/home



### INSTALLATION ###

# Move UK mirrors higher up the list
vim /etc/pacman.d/mirrorlist

# Install OS to mounted filesystem
pacstrap /mnt base base-devel

# Generate FS table
genfstab -U /mnt >> /mnt/etc/fstab
genfstab -U /mnt/home >> /mnt/etc/fstab

# Chroot into system
arch-chroot /mnt



### LOCALIZATION ###

ln -sf /usr/share/zoneinfo/Europe/London /etc/localtime
hwclock --systohc

echo "en_GB.UTF-8" >> /etc/locale.gen
locale-gen

echo "LANG=en_GB.UTF-8" > /etc/locale.conf
echo "KEYMAP=uk" > /etc/vconsole.conf



### HOSTNAME / HOSTS ###

echo "rupert" > /etc/hostname

echo "127.0.0.1       localhost
	127.0.1.1       rupert
	::1     localhost ip6-localhost ip6-loopback
	192.168.0.24    humphrey" > /etc/hosts



### USERS ###

# Set root password
passwd

useradd -m -g wheel jordan
passwd jordan

chmod 0640 /etc/sudoers
vi /etc/sudoers
# Uncomment "%wheel ALL=(ALL) ALL"



### BOOTLOADER

pacman -S grub
grub-install --target=i386-pc /dev/sda
vim /etc/default/grub #remove quiet
grub-mkconfig -o /boot/grub/grub.cfg



### NETWORKING ###

systemctl enable dhcpcd.service # is this needed when installing network manager?

systemctl enable NetworkManager.service



### KERNEL ENTROPY FIX ###

pacman -S haveged
